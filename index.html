    function calculateDeviceCost() {
    const status = document.getElementById('deviceStatus').value;
    const costInput = document.getElementById('deviceCost').value;
    const resultEl = document.getElementById('deviceResult');
    if (status === "none" || !costInput || costInput <= 0) {
    resultEl.classList.add('hidden');
    return;

    const cost = parseInt(costInput);
    let chargeRate = 0;
    let statusText = "";
    if (status === "intentional") {
    chargeRate = 1.0;
    statusText = "고의 파손 (100% 본인 부담)";
    } else if (status === "negligent") {
    chargeRate = 0.2;
    statusText = "과실 파손 (20% 본인 부담)";
    } else if (status === "lost") {
    chargeRate = 0.4;
    statusText = "단말기 분실 (40% 본인 부담)";
    }
    const clientCost = Math.floor(cost * chargeRate);
    resultEl.innerHTML = `
    <div class="space-y-1">
    <span class="block text-[10px] text-red-500 uppercase font-bold">${statusText}</span>
    <p class="text-sm">실제 사용자 부담 금액: <strong>${clientCost.toLocaleString()} 원</strong></p>
    </div>
    `;
    resultEl.classList.remove('hidden');
    }
